  const popup = document.getElementById('popup');
  const character = document.getElementById("character");
  const scoreDisplay = document.getElementById("score");
  const restartBtn = document.getElementById("restart-button");
  const playArea = document.getElementById("play-area");

  let isJumping = false;
  let score = 0;
  let interval;
  let obstacleInterval;
  let gameRunning = false;

  function enterSite() {
    document.getElementById("intro-screen").style.display = "none";
    document.getElementById("main-content").style.display = "flex";
    popup.style.display = 'block';
    setTimeout(() => popup.style.display = 'none', 3000);
  }

  function playSound() {
    document.getElementById('sound').play();
  }

  function startGame() {
    document.getElementById("main-content").style.display = "none";
    document.getElementById("game-screen").style.display = "flex";
    playArea.style.display = "block";
    score = 0;
    scoreDisplay.textContent = "åˆ†æ•¸: 0";
    gameRunning = true;
    interval = setInterval(() => {
      score++;
      scoreDisplay.textContent = "åˆ†æ•¸: " + score;
    }, 200);
    obstacleInterval = setInterval(createObstacle, 2500); // æ¯ 2.5 ç§’å‡ºç¾
  }

  function jump() {
    if (isJumping) return;
    isJumping = true;
    let pos = 0;
    const upInterval = setInterval(() => {
      if (pos >= 240) {  // æ›´é«˜ï¼
        clearInterval(upInterval);
        const downInterval = setInterval(() => {
          if (pos <= 0) {
            clearInterval(downInterval);
            isJumping = false;

            pos -= 12;
            character.style.bottom = pos + "px";
          }
        }, 12);
      } else {
        pos += 12;
        character.style.bottom = pos + "px";
      }
    }, 12);
  }

  function createObstacle() {
    if (!gameRunning) return;
    const obstacle = document.createElement("div");
    obstacle.classList.add("obstacle");
    let obstacleLeft = 800;
    obstacle.style.left = obstacleLeft + "px";
    playArea.appendChild(obstacle);

    const moveInterval = setInterval(() => {
      if (!gameRunning) {
        clearInterval(moveInterval);
        return;
      }
      if (obstacleLeft < 0) {
        playArea.removeChild(obstacle);
        clearInterval(moveInterval);
      } else {
        obstacleLeft -= 5;
        obstacle.style.left = obstacleLeft + "px";

        const charBottom = parseInt(character.style.bottom.replace("px", "")) || 0;
        if (obstacleLeft < 100 && obstacleLeft > 40 && charBottom < 60) {
          gameOver();
          clearInterval(moveInterval);
        }
      }
    }, 20);
  }

  function gameOver() {

    clearInterval(interval);
    clearInterval(obstacleInterval);
